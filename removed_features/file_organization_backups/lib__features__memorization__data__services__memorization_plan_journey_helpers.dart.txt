part of 'memorization_plan_journey_engine.dart';

extension _MemorizationPlanJourneyHelpers on MemorizationPlanJourneyEngine {
  int _effectiveReviewEveryForPlan({
    required MemorizationActivePlanModel plan,
    required int primarySessionsCount,
  }) {
    if (!plan.smartReviewDistribution) {
      return plan.reviewEveryDays.clamp(1, 7).toInt();
    }

    final sessions = primarySessionsCount <= 0
        ? plan.totalDays
        : primarySessionsCount;

    if (plan.reviewFromLearnedOnly) {
      if (sessions <= 10) return 2;
      if (sessions <= 30) return 3;
      return 4;
    }

    final reviewSessions = plan.reviewSessionsCount <= 0
        ? 1
        : plan.reviewSessionsCount;
    final every = (sessions / (reviewSessions + 1)).floor();
    return every.clamp(1, 6).toInt();
  }

  bool _isLearningTask(MemorizationJourneyTask item) {
    return item.task.type == 'dailyNew' || item.task.type == 'dailyReview';
  }

  bool _isFinalTest(MemorizationJourneyTask item) {
    return item.task.type == 'selfTest' &&
        (item.task.title.contains('ختام') ||
            item.task.subtitle.contains('ختام'));
  }

  bool _shouldReplaceSameDayTest(
    MemorizationJourneyTask existing,
    MemorizationJourneyTask candidate,
  ) {
    final existingMandatory = existing.timeLabel.contains('إلزامي');
    final candidateMandatory = candidate.timeLabel.contains('إلزامي');

    if (candidateMandatory != existingMandatory) return candidateMandatory;

    if (candidate.priority != existing.priority) {
      return candidate.priority < existing.priority;
    }

    return candidate.task.ayahsCount > existing.task.ayahsCount;
  }

  DateTime _journeyDateOnly(DateTime value) {
    return DateTime(value.year, value.month, value.day);
  }

  String _journeyDateKey(DateTime date) {
    final clean = _journeyDateOnly(date);
    return '${clean.year}-${clean.month}-${clean.day}';
  }

  bool _isWeeklyRestDay(DateTime date, int weeklyRestDays) {
    final count = weeklyRestDays.clamp(0, 3).toInt();
    if (count <= 0) return false;

    // توزيع ثابت وهادئ بدون أيام متتالية:
    // 1 يوم: الجمعة
    // 2 يوم: الثلاثاء والجمعة
    // 3 أيام: الاثنين والأربعاء والجمعة
    final weekday = date.weekday; // Monday = 1 ... Sunday = 7

    switch (count) {
      case 1:
        return weekday == DateTime.friday;
      case 2:
        return weekday == DateTime.tuesday || weekday == DateTime.friday;
      case 3:
        return weekday == DateTime.monday ||
            weekday == DateTime.wednesday ||
            weekday == DateTime.friday;
      default:
        return false;
    }
  }

  String _primaryTypeForAction(String action, String fallbackType) {
    if (action == 'newMemorization' || action == 'newWithReview') {
      return 'dailyNew';
    }

    if (action == 'reviewOnly' || action == 'strengthenAndTest') {
      return 'dailyReview';
    }

    return fallbackType.trim().isEmpty ? 'dailyNew' : fallbackType;
  }

  _TestVariation _testVariation(int index) {
    final value = index % 3;

    if (value == 0) {
      return const _TestVariation(
        code: 'starts',
        subtitle: 'اختبار بدايات الآيات. الموضع يظهر عند فتح الاختبار.',
      );
    }

    if (value == 1) {
      return const _TestVariation(
        code: 'hidden',
        subtitle: 'اختبار آيات مخفية. الموضع يظهر عند فتح الاختبار.',
      );
    }

    return const _TestVariation(
      code: 'page',
      subtitle: 'اختبار صفحة كاملة. الموضع يظهر عند فتح الاختبار.',
    );
  }

  int _estimateMinutes({required int ayahsCount, required String type}) {
    final multiplier = type == 'dailyNew'
        ? 1.35
        : type == 'selfTest'
        ? 0.55
        : 0.75;

    final minutes = 5 + (ayahsCount * multiplier);

    return minutes.clamp(6, 60).round();
  }

  DateTime _dateOnly(DateTime value) {
    return DateTime(value.year, value.month, value.day);
  }

  bool _sameDay(DateTime a, DateTime b) {
    return a.year == b.year && a.month == b.month && a.day == b.day;
  }
}
