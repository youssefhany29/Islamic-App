class _SelfEvaluationPanel extends StatelessWidget {
  const _SelfEvaluationPanel({
    required this.isAnswered,
    required this.isCorrect,
    required this.onExcellentTap,
    required this.onGoodTap,
    required this.onReviewTap,
    required this.onMistakesTap,
  });

  final bool isAnswered;
  final bool isCorrect;
  final VoidCallback onExcellentTap;
  final VoidCallback onGoodTap;
  final VoidCallback onReviewTap;
  final VoidCallback onMistakesTap;

  @override
  Widget build(BuildContext context) {
    final theme = Theme.of(context);

    if (isAnswered) {
      return Container(
        width: double.infinity,
        padding: EdgeInsets.all(12.w),
        decoration: BoxDecoration(
          color: isCorrect
              ? Colors.green.withOpacity(0.10)
              : theme.colorScheme.error.withOpacity(0.10),
          borderRadius: BorderRadius.circular(18.r),
        ),
        child: Text(
          isCorrect
              ? 'تم تسجيل السؤال كموضع محفوظ.'
              : 'تم تسجيل السؤال كموضع يحتاج تثبيت.',
          textDirection: TextDirection.rtl,
          textAlign: TextAlign.right,
          maxLines: 3,
          softWrap: true,
          overflow: TextOverflow.ellipsis,
          style: AppTextStyles.caption(context).copyWith(
            fontWeight: FontWeight.w900,
            color: isCorrect ? Colors.green.shade700 : theme.colorScheme.error,
            height: 1.5,
          ),
        ),
      );
    }

    return Column(
      crossAxisAlignment: CrossAxisAlignment.end,
      children: [
        Text(
          'بعد ما راجعت النص، قيّم تسميعك:',
          textDirection: TextDirection.rtl,
          textAlign: TextAlign.right,
          maxLines: 3,
          softWrap: true,
          overflow: TextOverflow.ellipsis,
          style: AppTextStyles.caption(context).copyWith(
            fontWeight: FontWeight.w900,
            color: theme.colorScheme.surface,
          ),
        ),
        SizedBox(height: 10.h),
        Row(
          children: [
            Expanded(
              child: _EvaluationButton(
                label: 'أخطأت في مواضع',
                icon: Icons.close_rounded,
                onTap: onMistakesTap,
                color: theme.colorScheme.error,
              ),
            ),
            SizedBox(width: 8.w),
            Expanded(
              child: _EvaluationButton(
                label: 'محتاج مراجعة',
                icon: Icons.help_outline_rounded,
                onTap: onReviewTap,
                color: Colors.orange,
              ),
            ),
          ],
        ),
        SizedBox(height: 8.h),
        Row(
          children: [
            Expanded(
              child: _EvaluationButton(
                label: 'جيد',
                icon: Icons.thumb_up_alt_rounded,
                onTap: onGoodTap,
                color: Colors.teal,
              ),
            ),
            SizedBox(width: 8.w),
            Expanded(
              child: _EvaluationButton(
                label: 'ممتاز',
                icon: Icons.workspace_premium_rounded,
                onTap: onExcellentTap,
                color: Colors.green,
              ),
            ),
          ],
        ),
      ],
    );
  }
}

class _EvaluationButton extends StatelessWidget {
  const _EvaluationButton({
    required this.label,
    required this.icon,
    required this.onTap,
    required this.color,
  });

  final String label;
  final IconData icon;
  final VoidCallback onTap;
  final Color color;

  @override
  Widget build(BuildContext context) {
    return Material(
      color: color.withOpacity(0.10),
      borderRadius: BorderRadius.circular(16.r),
      child: InkWell(
        onTap: onTap,
        borderRadius: BorderRadius.circular(16.r),
        child: Padding(
          padding: EdgeInsets.symmetric(horizontal: 6.w, vertical: 11.h),
          child: Row(
            mainAxisAlignment: MainAxisAlignment.center,
            textDirection: TextDirection.rtl,
            children: [
              Icon(icon, size: 15.sp, color: color),
              SizedBox(width: 5.w),
              Flexible(
                child: Text(
                  label,
                  textDirection: TextDirection.rtl,
                  textAlign: TextAlign.center,
                  maxLines: 3,
                  softWrap: true,
                  overflow: TextOverflow.ellipsis,
                  style: AppTextStyles.caption(
                    context,
                  ).copyWith(fontWeight: FontWeight.w900, color: color),
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

class _TestButton extends StatelessWidget {
  const _TestButton({
    required this.label,
    required this.icon,
    required this.onTap,
    required this.isPrimary,
    this.isDisabled = false,
  });

  final String label;
  final IconData icon;
  final VoidCallback onTap;
  final bool isPrimary;
  final bool isDisabled;

  @override
  Widget build(BuildContext context) {
    final theme = Theme.of(context);

    final backgroundColor = isPrimary
        ? theme.colorScheme.primary
        : theme.colorScheme.secondary;

    final foregroundColor = isPrimary
        ? Colors.white
        : theme.colorScheme.primary;

    return Opacity(
      opacity: isDisabled ? 0.45 : 1,
      child: Material(
        color: backgroundColor,
        borderRadius: BorderRadius.circular(18.r),
        child: InkWell(
          borderRadius: BorderRadius.circular(18.r),
          onTap: isDisabled ? null : onTap,
          child: Padding(
            padding: EdgeInsets.symmetric(horizontal: 6.w, vertical: 13.h),
            child: Row(
              mainAxisAlignment: MainAxisAlignment.center,
              textDirection: TextDirection.rtl,
              children: [
                Icon(icon, color: foregroundColor, size: 18.sp),
                SizedBox(width: 7.w),
                Flexible(
                  child: Text(
                    label,
                    textDirection: TextDirection.rtl,
                    textAlign: TextAlign.center,
                    maxLines: 3,
                    softWrap: true,
                    overflow: TextOverflow.ellipsis,
                    style: AppTextStyles.caption(context).copyWith(
                      fontWeight: FontWeight.w900,
                      color: foregroundColor,
                    ),
                  ),
                ),
              ],
            ),
          ),
        ),
      ),
    );
  }
}
