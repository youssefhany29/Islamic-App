part of 'memorization_plan_journey_engine.dart';

class MemorizationTestDayAllocator {
  const MemorizationTestDayAllocator();

  List<MemorizationJourneyTask> allocate({
    required List<MemorizationJourneyTask> tasks,
    required MemorizationPlanIntensity intensity,
    required int weeklyRestDays,
  }) {
    if (tasks.isEmpty) return const [];

    final learningByDate = <String, List<MemorizationJourneyTask>>{};
    final otherByDate = <String, List<MemorizationJourneyTask>>{};
    final tests = <MemorizationJourneyTask>[];

    for (final item in tasks) {
      final key = _dateKey(item.date);
      if (item.task.type == 'selfTest') {
        tests.add(item);
      } else if (_isNewMemorization(item)) {
        learningByDate.putIfAbsent(key, () => []).add(item);
      } else {
        otherByDate.putIfAbsent(key, () => []).add(item);
      }
    }

    final uniquePreferredTests = <String, MemorizationJourneyTask>{};
    for (final test in tests) {
      final key = _dateKey(test.date);
      final existing = uniquePreferredTests[key];
      if (existing == null || _prefer(test, existing)) {
        uniquePreferredTests[key] = test;
      }
    }

    final sortedTests = uniquePreferredTests.values.toList()
      ..sort((a, b) => a.date.compareTo(b.date));
    final allocatedTestDates = <String>{};
    final allocatedTests = <MemorizationJourneyTask>[];
    final finalTests = sortedTests.where(_isFinal).toList();
    final checkpointTests = sortedTests.where((item) => !_isFinal(item));

    for (final test in checkpointTests) {
      final allocatedDate = _nearestSuitableDate(
        preferred: _dateOnly(test.date),
        weeklyRestDays: weeklyRestDays,
        allocatedTestDates: allocatedTestDates,
      );
      allocatedTestDates.add(_dateKey(allocatedDate));
      allocatedTests.add(_copyWithDate(test, allocatedDate));
    }

    final rescheduledLearning = _rescheduleLearningDays(
      learningByDate: learningByDate,
      allocatedTestDates: allocatedTestDates,
      weeklyRestDays: weeklyRestDays,
    );

    DateTime? lastLearningDate;
    for (final item in rescheduledLearning) {
      if (lastLearningDate == null || item.date.isAfter(lastLearningDate)) {
        lastLearningDate = item.date;
      }
    }

    for (final test in finalTests) {
      final preferred = lastLearningDate == null
          ? _dateOnly(test.date)
          : _dateOnly(lastLearningDate).add(const Duration(days: 1));
      final allocatedDate = _nearestSuitableDate(
        preferred: preferred,
        weeklyRestDays: weeklyRestDays,
        allocatedTestDates: allocatedTestDates,
      );
      allocatedTestDates.add(_dateKey(allocatedDate));
      allocatedTests.add(_copyWithDate(test, allocatedDate));
    }

    return [
      for (final entries in otherByDate.values) ...entries,
      ...rescheduledLearning,
      ...allocatedTests,
    ];
  }

  DateTime _nearestSuitableDate({
    required DateTime preferred,
    required int weeklyRestDays,
    required Set<String> allocatedTestDates,
  }) {
    for (int offset = 0; offset <= 365; offset++) {
      final date = _dateOnly(preferred).add(Duration(days: offset));
      final key = _dateKey(date);
      if (allocatedTestDates.contains(key)) continue;
      if (_isRestDay(date, weeklyRestDays)) continue;
      return date;
    }

    var date = _dateOnly(preferred).add(const Duration(days: 366));
    while (allocatedTestDates.contains(_dateKey(date)) ||
        _isRestDay(date, weeklyRestDays)) {
      date = date.add(const Duration(days: 1));
    }
    return date;
  }

  List<MemorizationJourneyTask> _rescheduleLearningDays({
    required Map<String, List<MemorizationJourneyTask>> learningByDate,
    required Set<String> allocatedTestDates,
    required int weeklyRestDays,
  }) {
    final groups = learningByDate.values.toList()
      ..sort((a, b) => a.first.date.compareTo(b.first.date));
    final occupiedLearningDates = <String>{};
    final result = <MemorizationJourneyTask>[];

    for (final group in groups) {
      var date = _dateOnly(group.first.date);
      while (allocatedTestDates.contains(_dateKey(date)) ||
          occupiedLearningDates.contains(_dateKey(date)) ||
          _isRestDay(date, weeklyRestDays)) {
        date = date.add(const Duration(days: 1));
      }

      occupiedLearningDates.add(_dateKey(date));
      for (final item in group) {
        result.add(_copyWithDate(item, date));
      }
    }
    return result;
  }

  bool _prefer(
    MemorizationJourneyTask candidate,
    MemorizationJourneyTask existing,
  ) {
    final candidateMandatory = candidate.timeLabel.contains('إلزامي');
    final existingMandatory = existing.timeLabel.contains('إلزامي');
    if (candidateMandatory != existingMandatory) return candidateMandatory;
    return candidate.task.ayahsCount > existing.task.ayahsCount;
  }

  bool _isNewMemorization(MemorizationJourneyTask item) {
    return item.task.type == 'dailyNew';
  }

  bool _isFinal(MemorizationJourneyTask item) {
    return item.task.title.contains('ختام') ||
        item.task.subtitle.contains('ختام');
  }

  MemorizationJourneyTask _copyWithDate(
    MemorizationJourneyTask item,
    DateTime date,
  ) {
    final clean = _dateOnly(date);
    return MemorizationJourneyTask(
      task: item.task.copyWith(scheduledDate: clean, updatedAt: DateTime.now()),
      date: clean,
      timeLabel: item.timeLabel,
      priority: item.priority,
      isProjected: !_sameDay(clean, _dateOnly(DateTime.now())),
    );
  }

  bool _isRestDay(DateTime date, int weeklyRestDays) {
    final count = weeklyRestDays.clamp(0, 3).toInt();
    if (count == 1) return date.weekday == DateTime.friday;
    if (count == 2) {
      return date.weekday == DateTime.tuesday ||
          date.weekday == DateTime.friday;
    }
    if (count == 3) {
      return date.weekday == DateTime.monday ||
          date.weekday == DateTime.wednesday ||
          date.weekday == DateTime.friday;
    }
    return false;
  }

  DateTime _dateOnly(DateTime value) {
    return DateTime(value.year, value.month, value.day);
  }

  String _dateKey(DateTime date) {
    final clean = _dateOnly(date);
    return '${clean.year}-${clean.month}-${clean.day}';
  }

  bool _sameDay(DateTime a, DateTime b) {
    return a.year == b.year && a.month == b.month && a.day == b.day;
  }
}
