import 'dart:math' as math;

import '../models/memorization_active_plan_model.dart';
import '../models/memorization_session_result_model.dart';
import '../models/memorization_today_task_model.dart';
import '../services/memorization_session_result_storage.dart';
import 'memorization_balanced_range_splitter.dart';
import 'memorization_plan_progress_resolver.dart';
import 'memorization_smart_test_planner.dart';
import 'quran_memorization_range_resolver.dart';

class MemorizationJourneyTask {
  final MemorizationTodayTaskModel task;
  final DateTime date;
  final String timeLabel;
  final int priority;
  final bool isProjected;

  const MemorizationJourneyTask({
    required this.task,
    required this.date,
    required this.timeLabel,
    required this.priority,
    required this.isProjected,
  });

  bool get isToday {
    final now = DateTime.now();
    return date.year == now.year &&
        date.month == now.month &&
        date.day == now.day;
  }
}

class MemorizationPlanJourneyEngine {
  const MemorizationPlanJourneyEngine({
    this.splitter = const MemorizationBalancedRangeSplitter(),
    this.rangeResolver = const QuranMemorizationRangeResolver(),
    this.progressResolver = const MemorizationPlanProgressResolver(),
    this.smartTestPlanner = const MemorizationSmartTestPlanner(),
  });

  final MemorizationBalancedRangeSplitter splitter;
  final QuranMemorizationRangeResolver rangeResolver;
  final MemorizationPlanProgressResolver progressResolver;
  final MemorizationSmartTestPlanner smartTestPlanner;

  Future<List<MemorizationJourneyTask>> buildJourneyTasks({
    required MemorizationActivePlanModel? plan,
    required MemorizationTodayTaskModel? activeTask,
    int daysAhead = 7,
  }) async {
    await rangeResolver.ensureReady();

    if (plan == null || !plan.hasValidScopeRange) return const [];

    final today = _dateOnly(DateTime.now());
    final results = await MemorizationSessionResultStorage.getResults();

    final progress = progressResolver.resolve(
      plan: plan,
      results: results,
      today: today,
    );

    final action = plan.actionTypeName;
    final primaryType = _primaryTypeForAction(
      action,
      activeTask?.type ?? 'dailyNew',
    );

    final primaryChunks = await _buildPrimaryChunks(
      plan: plan,
      primaryType: primaryType,
    );

    final completedPrimaryCount = primaryType == 'dailyNew'
        ? progress.completedNewCount
        : progress.completedReviewCount;

    final safeDaysAhead = math.max(0, daysAhead);

    final tasks = await _buildOneLearningTaskPerDay(
      plan: plan,
      action: action,
      primaryType: primaryType,
      primaryChunks: primaryChunks,
      progress: progress,
      completedPrimaryCount: completedPrimaryCount,
      safeDaysAhead: safeDaysAhead,
    );

    final plannedSmartTests = smartTestPlanner.buildTests(
      plan: plan,
      sourceTasks: tasks
          .where(
            (item) =>
        item.task.type == 'dailyNew' ||
            item.task.type == 'dailyReview',
      )
          .map(
            (item) => MemorizationTestSourceTask(
          task: item.task,
          date: item.date,
          priority: item.priority,
        ),
      )
          .toList(),
      results: results,
      daysAhead: safeDaysAhead,
    );

    // نسمح بهامش بعد مدة الحفظ/المراجعة لأن أيام الاختبار تفصل المهام
    // وقد تزق ختام الدورة بعد آخر مهمة فعلية.
    final int testDaysBuffer = math.max(14, (safeDaysAhead / 4).ceil());
    final lastAllowedDate =
    today.add(Duration(days: safeDaysAhead + testDaysBuffer));

    for (final test in plannedSmartTests) {
      final testDate = _dateOnly(test.scheduledDate);

      if (testDate.isBefore(today) || testDate.isAfter(lastAllowedDate)) {
        continue;
      }

      tasks.add(
        MemorizationJourneyTask(
          task: test.toTodayTask(),
          date: testDate,
          timeLabel: test.isMandatory ? 'اختبار إلزامي' : 'اختبار',
          priority: test.isMandatory ? 3 : 4,
          isProjected: !_sameDay(testDate, today),
        ),
      );
    }

    final rescheduledTasks = _moveLearningTasksAwayFromTestDays(
      tasks,
      weeklyRestDays: plan.weeklyRestDays,
    );

    rescheduledTasks.sort((a, b) {
      final dateCompare = a.date.compareTo(b.date);
      if (dateCompare != 0) return dateCompare;
      return a.priority.compareTo(b.priority);
    });

    return rescheduledTasks;
  }

  Future<List<MemorizationJourneyTask>> _buildOneLearningTaskPerDay({
    required MemorizationActivePlanModel plan,
    required String action,
    required String primaryType,
    required List<MemorizationScheduleChunk> primaryChunks,
    required MemorizationPlanProgressSnapshot progress,
    required int completedPrimaryCount,
    required int safeDaysAhead,
  }) async {
    if (primaryChunks.isEmpty) return const [];

    final today = _dateOnly(DateTime.now());
    final tasks = <MemorizationJourneyTask>[];

    final reviewChunks = await _buildIndependentReviewChunks(plan);
    final effectiveReviewEvery = _effectiveReviewEveryForPlan(
      plan: plan,
      primarySessionsCount: primaryChunks.length,
    );

    int primaryIndex =
    completedPrimaryCount.clamp(0, primaryChunks.length).toInt();

    int reviewIndex = progress.completedReviewCount.clamp(0, 999999).toInt();
    int learnedReviewIndex =
    progress.completedReviewCount.clamp(0, 999999).toInt();

    int dayOffset = 0;
    int learningDaysUsed = 0;

    int lastIndependentReviewPrimaryCount = -1;
    int lastIndependentReviewDayOffset = -999;
    bool externalComprehensiveReviewDone = false;

    while (dayOffset <= safeDaysAhead) {
      final date = today.add(Duration(days: dayOffset));

      if (_isWeeklyRestDay(date, plan.weeklyRestDays)) {
        dayOffset++;
        continue;
      }

      MemorizationJourneyTask? nextTask;

      if (action == 'newWithReview') {
        if (plan.reviewFromLearnedOnly) {
          final learnedCount = primaryIndex;

          final shouldReviewLearned = _shouldScheduleLearnedReviewDay(
            learningDaysUsed: learningDaysUsed,
            learnedCount: learnedCount,
            reviewIndex: learnedReviewIndex,
            hasMoreNew: primaryIndex < primaryChunks.length,
            every: effectiveReviewEvery,
          );

          if (shouldReviewLearned) {
            nextTask = _learnedReviewTaskForDay(
              plan: plan,
              primaryChunks: primaryChunks,
              learnedCount: learnedCount,
              reviewIndex: learnedReviewIndex,
              date: date,
              isProjected: dayOffset != 0,
            );

            learnedReviewIndex++;
            reviewIndex++;
            lastIndependentReviewPrimaryCount = primaryIndex;
            lastIndependentReviewDayOffset = dayOffset;
          } else if (primaryIndex < primaryChunks.length) {
            final chunk = primaryChunks[primaryIndex];

            nextTask = _primaryJourneyTaskFromChunk(
              plan: plan,
              chunk: chunk,
              index: primaryIndex,
              type: primaryType,
              date: date,
              isProjected: dayOffset != 0,
            );

            primaryIndex++;
          } else if (_shouldAddFinalLearnedReview(
            learnedCount: learnedCount,
            reviewIndex: learnedReviewIndex,
          )) {
            nextTask = _learnedReviewTaskForDay(
              plan: plan,
              primaryChunks: primaryChunks,
              learnedCount: learnedCount,
              reviewIndex: learnedReviewIndex,
              date: date,
              isProjected: dayOffset != 0,
              forceCumulative: true,
            );

            learnedReviewIndex++;
            reviewIndex++;
          }
        } else {
          final shouldReviewExternal = _shouldScheduleIndependentReviewDay(
            primaryDaysDone: primaryIndex,
            reviewIndex: reviewIndex,
            reviewChunksCount: reviewChunks.length,
            every: effectiveReviewEvery,
            hasMoreNew: primaryIndex < primaryChunks.length,
            currentDayOffset: dayOffset,
            lastReviewPrimaryCount: lastIndependentReviewPrimaryCount,
            lastReviewDayOffset: lastIndependentReviewDayOffset,
          );

          if (shouldReviewExternal) {
            final chunk = reviewChunks[reviewIndex];

            nextTask = _reviewJourneyTaskFromChunk(
              plan: plan,
              chunk: chunk,
              index: reviewIndex,
              date: date,
              isProjected: dayOffset != 0,
              title: 'مراجعة اليوم',
              subtitle: 'مراجعة ${chunk.ayahsCount} آية من ${plan.reviewScopeTitle}',
              timeCode: 'reviewSeparate',
              minGlobalAyahIndex: plan.reviewStartGlobalAyahIndex,
              maxGlobalAyahIndex: plan.reviewEndGlobalAyahIndex,
            );

            reviewIndex++;
            lastIndependentReviewPrimaryCount = primaryIndex;
            lastIndependentReviewDayOffset = dayOffset;
          } else if (primaryIndex < primaryChunks.length) {
            final chunk = primaryChunks[primaryIndex];

            nextTask = _primaryJourneyTaskFromChunk(
              plan: plan,
              chunk: chunk,
              index: primaryIndex,
              type: primaryType,
              date: date,
              isProjected: dayOffset != 0,
            );

            primaryIndex++;
          } else if (reviewIndex < reviewChunks.length) {
            final chunk = reviewChunks[reviewIndex];

            nextTask = _reviewJourneyTaskFromChunk(
              plan: plan,
              chunk: chunk,
              index: reviewIndex,
              date: date,
              isProjected: dayOffset != 0,
              title: 'مراجعة اليوم',
              subtitle: 'مراجعة ${chunk.ayahsCount} آية من ${plan.reviewScopeTitle}',
              timeCode: 'reviewSeparate',
              minGlobalAyahIndex: plan.reviewStartGlobalAyahIndex,
              maxGlobalAyahIndex: plan.reviewEndGlobalAyahIndex,
            );

            reviewIndex++;
            lastIndependentReviewPrimaryCount = primaryIndex;
            lastIndependentReviewDayOffset = dayOffset;
          } else if (!externalComprehensiveReviewDone && reviewChunks.isNotEmpty) {
            nextTask = _externalComprehensiveReviewTask(
              plan: plan,
              date: date,
              isProjected: dayOffset != 0,
            );

            externalComprehensiveReviewDone = true;
            lastIndependentReviewPrimaryCount = primaryIndex;
            lastIndependentReviewDayOffset = dayOffset;
          }
        }
      } else if (primaryIndex < primaryChunks.length) {
        final chunk = primaryChunks[primaryIndex];

        nextTask = _primaryJourneyTaskFromChunk(
          plan: plan,
          chunk: chunk,
          index: primaryIndex,
          type: primaryType,
          date: date,
          isProjected: dayOffset != 0,
        );

        primaryIndex++;
      }

      if (nextTask != null) {
        tasks.add(nextTask);
        learningDaysUsed++;
      }

      final primaryDone = primaryIndex >= primaryChunks.length;

      final reviewDone = action != 'newWithReview' ||
          (plan.reviewFromLearnedOnly
              ? primaryDone &&
              !_shouldAddFinalLearnedReview(
                learnedCount: primaryIndex,
                reviewIndex: learnedReviewIndex,
              )
              : reviewIndex >= reviewChunks.length &&
              (reviewChunks.isEmpty || externalComprehensiveReviewDone));

      if (primaryDone && reviewDone) break;

      dayOffset++;
    }

    return tasks;
  }

  int _effectiveReviewEveryForPlan({
    required MemorizationActivePlanModel plan,
    required int primarySessionsCount,
  }) {
    if (!plan.smartReviewDistribution) {
      return plan.reviewEveryDays.clamp(1, 7).toInt();
    }

    final sessions = primarySessionsCount <= 0 ? plan.totalDays : primarySessionsCount;

    if (plan.reviewFromLearnedOnly) {
      if (sessions <= 10) return 2;
      if (sessions <= 30) return 3;
      return 4;
    }

    final reviewSessions = plan.reviewSessionsCount <= 0 ? 1 : plan.reviewSessionsCount;
    final every = (sessions / (reviewSessions + 1)).floor();
    return every.clamp(1, 6).toInt();
  }

  Future<List<MemorizationScheduleChunk>> _buildIndependentReviewChunks(
      MemorizationActivePlanModel plan,
      ) async {
    if (!plan.hasValidReviewRange || plan.reviewFromLearnedOnly) {
      return const [];
    }

    return splitter.split(
      startGlobalAyahIndex: plan.reviewStartGlobalAyahIndex,
      endGlobalAyahIndex: plan.reviewEndGlobalAyahIndex,
      targetSessions: _reviewSessionsCountForPlan(plan),
      pagesPerSession: plan.dailyReviewPages > 0 ? plan.dailyReviewPages : null,
    );
  }

  bool _shouldScheduleLearnedReviewDay({
    required int learningDaysUsed,
    required int learnedCount,
    required int reviewIndex,
    required bool hasMoreNew,
    required int every,
  }) {
    if (learnedCount <= 0) return false;
    if (learningDaysUsed <= 0) return false;

    final safeEvery = every.clamp(1, 7).toInt();

    // مراجعة قريبة بعد عدد من جلسات الحفظ، وليس بعد عدد أيام خام.
    final bool closeReview = learnedCount >= safeEvery &&
        learningDaysUsed % (safeEvery + 1) == safeEvery;

    // مراجعة شاملة كل فترة على كل ما سبق.
    final int widerEvery = math.max(7, safeEvery * 4);
    final bool widerReview = learnedCount >= 7 &&
        learningDaysUsed % (widerEvery + 1) == widerEvery;

    if (widerReview) return true;
    if (closeReview) return true;

    return false;
  }

  bool _shouldAddFinalLearnedReview({
    required int learnedCount,
    required int reviewIndex,
  }) {
    if (learnedCount <= 0) return false;

    // لو الحفظ قليل، نضيف مراجعة نهائية واحدة.
    if (learnedCount < 4) return reviewIndex == 0;

    // نضمن وجود مراجعة ختامية لو آخر مراجعة لم تكن شاملة.
    return reviewIndex == 0 || reviewIndex % 4 != 0;
  }

  bool _shouldScheduleIndependentReviewDay({
    required int primaryDaysDone,
    required int reviewIndex,
    required int reviewChunksCount,
    required int every,
    required bool hasMoreNew,
    required int currentDayOffset,
    required int lastReviewPrimaryCount,
    required int lastReviewDayOffset,
  }) {
    if (reviewIndex >= reviewChunksCount) return false;

    final safeEvery = every.clamp(1, 7).toInt();

    if (hasMoreNew) {
      if (primaryDaysDone <= 0) return false;
      if (primaryDaysDone == lastReviewPrimaryCount) return false;
      return primaryDaysDone % safeEvery == 0;
    }

    // بعد انتهاء الحفظ لا نرمي كل المراجعات وراء بعض.
    // نكملها بنفس الإيقاع الذي اختاره المستخدم: كل يوم / كل 3 أيام / ...
    if (lastReviewDayOffset < 0) return true;
    return currentDayOffset - lastReviewDayOffset >= safeEvery;
  }

  MemorizationJourneyTask _primaryJourneyTaskFromChunk({
    required MemorizationActivePlanModel plan,
    required MemorizationScheduleChunk chunk,
    required int index,
    required String type,
    required DateTime date,
    required bool isProjected,
  }) {
    return MemorizationJourneyTask(
      task: _taskFromChunk(
        plan: plan,
        chunk: chunk,
        index: index,
        type: type,
        title: type == 'dailyNew' ? 'حفظ الصباح' : 'مراجعة اليوم',
        subtitle: type == 'dailyNew'
            ? 'حفظ ${chunk.ayahsCount} آية'
            : 'مراجعة ${chunk.ayahsCount} آية',
        scheduledDate: date,
        timeCode: type == 'dailyNew' ? 'morning' : 'mainReview',
        expectedMinutes: _estimateMinutes(
          ayahsCount: chunk.ayahsCount,
          type: type,
        ),
      ),
      date: date,
      timeLabel: type == 'dailyNew' ? 'الصباح' : 'اليوم',
      priority: type == 'dailyNew' ? 1 : 2,
      isProjected: isProjected,
    );
  }

  MemorizationJourneyTask _reviewJourneyTaskFromChunk({
    required MemorizationActivePlanModel plan,
    required MemorizationScheduleChunk chunk,
    required int index,
    required DateTime date,
    required bool isProjected,
    required String title,
    required String subtitle,
    required String timeCode,
    int? minGlobalAyahIndex,
    int? maxGlobalAyahIndex,
  }) {
    return MemorizationJourneyTask(
      task: _taskFromChunk(
        plan: plan,
        chunk: chunk,
        index: index,
        type: 'dailyReview',
        title: title,
        subtitle: subtitle,
        scheduledDate: date,
        timeCode: '${timeCode}_$index',
        expectedMinutes: _estimateMinutes(
          ayahsCount: chunk.ayahsCount,
          type: 'dailyReview',
        ),
        minGlobalAyahIndex: minGlobalAyahIndex,
        maxGlobalAyahIndex: maxGlobalAyahIndex,
      ),
      date: date,
      timeLabel: 'المراجعة',
      priority: 2,
      isProjected: isProjected,
    );
  }

  MemorizationJourneyTask _externalComprehensiveReviewTask({
    required MemorizationActivePlanModel plan,
    required DateTime date,
    required bool isProjected,
  }) {
    final start = plan.reviewStartGlobalAyahIndex;
    final end = plan.reviewEndGlobalAyahIndex < start
        ? start
        : plan.reviewEndGlobalAyahIndex;
    final count = end - start + 1;

    return MemorizationJourneyTask(
      task: _taskFromRange(
        plan: plan,
        type: 'dailyReview',
        title: 'مراجعة شاملة',
        subtitle: 'مراجعة نطاق المراجعة كاملًا',
        startGlobalAyahIndex: start,
        endGlobalAyahIndex: end,
        scheduledDate: date,
        timeCode: 'reviewSeparateComprehensive',
        expectedMinutes: _estimateMinutes(
          ayahsCount: count,
          type: 'dailyReview',
        ),
        minGlobalAyahIndex: plan.reviewStartGlobalAyahIndex,
        maxGlobalAyahIndex: plan.reviewEndGlobalAyahIndex,
      ),
      date: date,
      timeLabel: 'المراجعة',
      priority: 2,
      isProjected: isProjected,
    );
  }

  MemorizationJourneyTask? _learnedReviewTaskForDay({
    required MemorizationActivePlanModel plan,
    required List<MemorizationScheduleChunk> primaryChunks,
    required int learnedCount,
    required int reviewIndex,
    required DateTime date,
    required bool isProjected,
    bool forceCumulative = false,
  }) {
    if (primaryChunks.isEmpty || learnedCount <= 0) return null;

    final lastLearnedIndex =
    (learnedCount - 1).clamp(0, primaryChunks.length - 1).toInt();

    final bool cumulative = forceCumulative || reviewIndex % 4 == 3;

    final startIndex = cumulative
        ? 0
        : (lastLearnedIndex - 1).clamp(0, lastLearnedIndex).toInt();

    final firstChunk = primaryChunks[startIndex];
    final lastChunk = primaryChunks[lastLearnedIndex];

    final start = firstChunk.startGlobalAyahIndex;
    final end = lastChunk.endGlobalAyahIndex;
    final count = end - start + 1;

    return MemorizationJourneyTask(
      task: _taskFromRange(
        plan: plan,
        type: 'dailyReview',
        title: cumulative ? 'مراجعة شاملة' : 'مراجعة قريبة',
        subtitle:
        cumulative ? 'مراجعة كل ما حفظته حتى الآن' : 'تثبيت آخر ما حفظته',
        startGlobalAyahIndex: start,
        endGlobalAyahIndex: end,
        scheduledDate: date,
        timeCode: cumulative
            ? 'reviewLearnedCumulative_$reviewIndex'
            : 'reviewLearnedRecent_$reviewIndex',
        expectedMinutes: _estimateMinutes(
          ayahsCount: count,
          type: 'dailyReview',
        ),
        minGlobalAyahIndex: plan.scopeStartGlobalAyahIndex,
        maxGlobalAyahIndex: plan.scopeEndGlobalAyahIndex,
      ),
      date: date,
      timeLabel: 'المراجعة',
      priority: 2,
      isProjected: isProjected,
    );
  }

  /// يوم الاختبار = اختبار فقط.
  /// الحفظ/المراجعة لا يختفوا من الخطة، بل ينتقلوا لأول يوم مناسب بعد الاختبار.
  /// اختبار ختام الدورة يُنقل تلقائيًا بعد آخر مهمة حفظ/مراجعة فعلية بعد إعادة الجدولة.
  List<MemorizationJourneyTask> _moveLearningTasksAwayFromTestDays(
      List<MemorizationJourneyTask> tasks, {
        required int weeklyRestDays,
      }) {
    if (tasks.isEmpty) return const [];

    final sorted = List<MemorizationJourneyTask>.from(tasks)
      ..sort((a, b) {
        final dateCompare = a.date.compareTo(b.date);
        if (dateCompare != 0) return dateCompare;
        return a.priority.compareTo(b.priority);
      });

    final chosenTestByDate = <String, MemorizationJourneyTask>{};
    final learningTasks = <MemorizationJourneyTask>[];
    final otherTasks = <MemorizationJourneyTask>[];

    for (final item in sorted) {
      if (item.task.type == 'selfTest') {
        final key = _journeyDateKey(item.date);
        final existing = chosenTestByDate[key];

        if (existing == null || _shouldReplaceSameDayTest(existing, item)) {
          chosenTestByDate[key] = item;
        }

        continue;
      }

      if (_isLearningTask(item)) {
        learningTasks.add(item);
      } else {
        otherTasks.add(item);
      }
    }

    final reservedDates = <String>{
      ...chosenTestByDate.keys,
      ...otherTasks.map((item) => _journeyDateKey(item.date)),
    };

    final movedLearningTasks = <MemorizationJourneyTask>[];

    for (final item in learningTasks) {
      var newDate = _journeyDateOnly(item.date);

      while (reservedDates.contains(_journeyDateKey(newDate)) ||
          _isWeeklyRestDay(newDate, weeklyRestDays)) {
        newDate = newDate.add(const Duration(days: 1));
      }

      reservedDates.add(_journeyDateKey(newDate));

      movedLearningTasks.add(
        _copyJourneyTaskWithDate(
          item: item,
          date: newDate,
        ),
      );
    }

    final adjustedTests = <MemorizationJourneyTask>[];

    for (final test in chosenTestByDate.values) {
      if (!_isFinalTest(test)) {
        adjustedTests.add(test);
        continue;
      }

      var finalDate = _dateAfterLastLearningTask(
        movedLearningTasks,
        fallback: test.date,
      );

      while (reservedDates.contains(_journeyDateKey(finalDate)) ||
          _isWeeklyRestDay(finalDate, weeklyRestDays)) {
        finalDate = finalDate.add(const Duration(days: 1));
      }

      reservedDates.add(_journeyDateKey(finalDate));

      adjustedTests.add(
        _copyJourneyTaskWithDate(
          item: test,
          date: finalDate,
        ),
      );
    }

    return <MemorizationJourneyTask>[
      ...otherTasks,
      ...movedLearningTasks,
      ...adjustedTests,
    ];
  }

  bool _isLearningTask(MemorizationJourneyTask item) {
    return item.task.type == 'dailyNew' || item.task.type == 'dailyReview';
  }

  bool _isFinalTest(MemorizationJourneyTask item) {
    return item.task.type == 'selfTest' &&
        (item.task.title.contains('ختام') || item.task.subtitle.contains('ختام'));
  }

  DateTime _dateAfterLastLearningTask(
      List<MemorizationJourneyTask> learningTasks, {
        required DateTime fallback,
      }) {
    if (learningTasks.isEmpty) {
      return _journeyDateOnly(fallback);
    }

    var last = _journeyDateOnly(learningTasks.first.date);

    for (final item in learningTasks) {
      final day = _journeyDateOnly(item.date);
      if (day.isAfter(last)) last = day;
    }

    return last.add(const Duration(days: 1));
  }

  bool _shouldReplaceSameDayTest(
      MemorizationJourneyTask existing,
      MemorizationJourneyTask candidate,
      ) {
    final existingMandatory = existing.timeLabel.contains('إلزامي');
    final candidateMandatory = candidate.timeLabel.contains('إلزامي');

    if (candidateMandatory != existingMandatory) return candidateMandatory;

    if (candidate.priority != existing.priority) {
      return candidate.priority < existing.priority;
    }

    return candidate.task.ayahsCount > existing.task.ayahsCount;
  }

  MemorizationJourneyTask _copyJourneyTaskWithDate({
    required MemorizationJourneyTask item,
    required DateTime date,
  }) {
    final cleanDate = _journeyDateOnly(date);

    final movedTask = item.task.copyWith(
      scheduledDate: cleanDate,
      updatedAt: DateTime.now(),
    );

    return MemorizationJourneyTask(
      task: movedTask,
      date: cleanDate,
      timeLabel: item.timeLabel,
      priority: item.priority,
      isProjected: !_sameDay(cleanDate, _journeyDateOnly(DateTime.now())),
    );
  }

  DateTime _journeyDateOnly(DateTime value) {
    return DateTime(value.year, value.month, value.day);
  }

  String _journeyDateKey(DateTime date) {
    final clean = _journeyDateOnly(date);
    return '${clean.year}-${clean.month}-${clean.day}';
  }

  bool _isWeeklyRestDay(DateTime date, int weeklyRestDays) {
    final count = weeklyRestDays.clamp(0, 3).toInt();
    if (count <= 0) return false;

    // توزيع ثابت وهادئ بدون أيام متتالية:
    // 1 يوم: الجمعة
    // 2 يوم: الثلاثاء والجمعة
    // 3 أيام: الاثنين والأربعاء والجمعة
    final weekday = date.weekday; // Monday = 1 ... Sunday = 7

    switch (count) {
      case 1:
        return weekday == DateTime.friday;
      case 2:
        return weekday == DateTime.tuesday || weekday == DateTime.friday;
      case 3:
        return weekday == DateTime.monday ||
            weekday == DateTime.wednesday ||
            weekday == DateTime.friday;
      default:
        return false;
    }
  }

  String _primaryTypeForAction(String action, String fallbackType) {
    if (action == 'newMemorization' || action == 'newWithReview') {
      return 'dailyNew';
    }

    if (action == 'reviewOnly' || action == 'strengthenAndTest') {
      return 'dailyReview';
    }

    return fallbackType.trim().isEmpty ? 'dailyNew' : fallbackType;
  }

  Future<List<MemorizationScheduleChunk>> _buildPrimaryChunks({
    required MemorizationActivePlanModel plan,
    required String primaryType,
  }) {
    final isNew = primaryType == 'dailyNew';

    return splitter.split(
      startGlobalAyahIndex: plan.scopeStartGlobalAyahIndex,
      endGlobalAyahIndex: plan.scopeEndGlobalAyahIndex,
      targetSessions: _targetSessionsForPrimary(plan: plan, isNew: isNew),
      pagesPerSession: isNew
          ? (plan.dailyNewPages > 0 ? plan.dailyNewPages : null)
          : (plan.dailyReviewPages > 0 ? plan.dailyReviewPages : null),
    );
  }

  int _targetSessionsForPrimary({
    required MemorizationActivePlanModel plan,
    required bool isNew,
  }) {
    if (isNew) {
      if (plan.totalDays > 0) return plan.totalDays;

      if (plan.totalPages > 0 && plan.dailyNewPages > 0) {
        return (plan.totalPages / plan.dailyNewPages).ceil();
      }

      return 1;
    }

    final reviewSessions = _reviewSessionsCountForPlan(plan);
    if (reviewSessions > 0) return reviewSessions;

    if (plan.totalDays > 0) return plan.totalDays;

    if (plan.totalPages > 0 && plan.dailyReviewPages > 0) {
      return (plan.totalPages / plan.dailyReviewPages).ceil();
    }

    return 1;
  }

  Future<MemorizationJourneyTask?> _reviewTaskForOffset({
    required MemorizationActivePlanModel plan,
    required List<MemorizationScheduleChunk> primaryChunks,
    required MemorizationPlanProgressSnapshot progress,
    required int offset,
    required DateTime date,
  }) async {
    final every = plan.reviewEveryDays.clamp(1, 7).toInt();
    final firstReviewOffset = progress.completedReviewToday ? every : 0;

    if (offset < firstReviewOffset) return null;
    if ((offset - firstReviewOffset) % every != 0) return null;

    final reviewSlotIndex = (offset - firstReviewOffset) ~/ every;
    final reviewIndex = progress.completedReviewCount + reviewSlotIndex;

    if (plan.reviewFromLearnedOnly) {
      final projectedCompletedNew = math.max(
        progress.completedNewCount,
        progress.completedNewCount + offset,
      );

      final availableCount = math.min(
        projectedCompletedNew,
        primaryChunks.length,
      );

      if (availableCount <= 0) return null;

      final bool fullReviewDay = offset > 0 && offset % 14 == 0;

      if (fullReviewDay) {
        final start = primaryChunks.first.startGlobalAyahIndex;
        final end = primaryChunks[availableCount - 1].endGlobalAyahIndex;
        final ayahsCount = end - start + 1;

        return MemorizationJourneyTask(
          task: _taskFromRange(
            plan: plan,
            type: 'dailyReview',
            title: 'مراجعة شاملة',
            subtitle: 'مراجعة كل ما حفظته حتى الآن',
            startGlobalAyahIndex: start,
            endGlobalAyahIndex: end,
            scheduledDate: date,
            timeCode: 'reviewFull_$reviewIndex',
            expectedMinutes: _estimateMinutes(
              ayahsCount: ayahsCount,
              type: 'dailyReview',
            ),
          ),
          date: date,
          timeLabel: 'المساء',
          priority: 2,
          isProjected: offset != 0,
        );
      }

      final chunk = primaryChunks[reviewIndex % availableCount];

      return MemorizationJourneyTask(
        task: _taskFromChunk(
          plan: plan,
          chunk: chunk,
          index: reviewIndex,
          type: 'dailyReview',
          title: 'مراجعة المساء',
          subtitle: 'تثبيت ${chunk.ayahsCount} آية من آخر ما حفظته',
          scheduledDate: date,
          timeCode: 'reviewLearned',
          expectedMinutes: _estimateMinutes(
            ayahsCount: chunk.ayahsCount,
            type: 'dailyReview',
          ),
          minGlobalAyahIndex: plan.scopeStartGlobalAyahIndex,
          maxGlobalAyahIndex: plan.scopeEndGlobalAyahIndex,
        ),
        date: date,
        timeLabel: 'المساء',
        priority: 2,
        isProjected: offset != 0,
      );
    }

    if (!plan.hasValidReviewRange) return null;

    final reviewChunks = await splitter.split(
      startGlobalAyahIndex: plan.reviewStartGlobalAyahIndex,
      endGlobalAyahIndex: plan.reviewEndGlobalAyahIndex,
      targetSessions: _reviewSessionsCountForPlan(plan),
      pagesPerSession: plan.dailyReviewPages > 0 ? plan.dailyReviewPages : null,
    );

    if (reviewIndex >= reviewChunks.length) return null;

    final chunk = reviewChunks[reviewIndex];

    return MemorizationJourneyTask(
      task: _taskFromChunk(
        plan: plan,
        chunk: chunk,
        index: reviewIndex,
        type: 'dailyReview',
        title: 'مراجعة المساء',
        subtitle: 'مراجعة ${chunk.ayahsCount} آية من ${plan.reviewScopeTitle}',
        scheduledDate: date,
        timeCode: 'reviewSeparate',
        expectedMinutes: _estimateMinutes(
          ayahsCount: chunk.ayahsCount,
          type: 'dailyReview',
        ),
        minGlobalAyahIndex: plan.reviewStartGlobalAyahIndex,
        maxGlobalAyahIndex: plan.reviewEndGlobalAyahIndex,
      ),
      date: date,
      timeLabel: 'المساء',
      priority: 2,
      isProjected: offset != 0,
    );
  }

  MemorizationJourneyTask? _testTaskForOffset({
    required MemorizationActivePlanModel plan,
    required String action,
    required List<MemorizationScheduleChunk> primaryChunks,
    required int completedPrimaryCount,
    required int completedSelfTestCount,
    required bool completedSelfTestToday,
    required int offset,
    required DateTime date,
  }) {
    if (action != 'strengthenAndTest' && plan.plannedTestsCount <= 0) {
      return null;
    }

    final testsCount = _plannedTestsForPlan(plan: plan);
    if (testsCount <= 0) return null;
    if (completedSelfTestCount >= testsCount) return null;
    if (primaryChunks.isEmpty) return null;

    final interval = math.max(
      2,
      (math.max(1, primaryChunks.length) / testsCount).ceil(),
    );

    final firstEligibleOffset = completedSelfTestToday ? 1 : 1;
    if (offset < firstEligibleOffset) return null;

    final shiftedOffset = offset - firstEligibleOffset;
    if (shiftedOffset % interval != 0) return null;

    final testIndex = completedSelfTestCount + (shiftedOffset ~/ interval);
    if (testIndex >= testsCount) return null;

    final completedBasedIndex = completedPrimaryCount > 0
        ? math.min(
      completedPrimaryCount - 1,
      math.max(
        0,
        ((testIndex + 1) * completedPrimaryCount / testsCount).ceil() - 1,
      ),
    )
        : math.max(
      0,
      ((testIndex + 1) * primaryChunks.length / testsCount).ceil() - 1,
    );

    final safeChunkIndex = completedBasedIndex
        .clamp(0, primaryChunks.length - 1)
        .toInt();

    final sourceChunk = primaryChunks[safeChunkIndex];
    final testRange = _testRangeFromChunk(
      chunk: sourceChunk,
      testIndex: testIndex,
    );
    final variation = _testVariation(testIndex);

    return MemorizationJourneyTask(
      task: _taskFromRange(
        plan: plan,
        type: 'selfTest',
        title: 'اختبار تثبيت ${testIndex + 1}',
        subtitle: variation.subtitle,
        startGlobalAyahIndex: testRange.start,
        endGlobalAyahIndex: testRange.end,
        scheduledDate: date,
        timeCode: 'test_${variation.code}_$testIndex',
        expectedMinutes: _estimateMinutes(
          ayahsCount: testRange.ayahsCount,
          type: 'selfTest',
        ),
      ),
      date: date,
      timeLabel: 'اختبار',
      priority: 3,
      isProjected: offset != 0,
    );
  }

  int _plannedTestsForPlan({required MemorizationActivePlanModel plan}) {
    if (plan.plannedTestsCount > 0) return plan.plannedTestsCount;

    final totalAyahs =
        plan.scopeEndGlobalAyahIndex - plan.scopeStartGlobalAyahIndex + 1;
    final totalPages = plan.totalPages;

    if (totalAyahs <= 10 || totalPages <= 3) return 1;
    if (totalPages <= 10) return 2;
    if (totalPages <= 30) return 3;
    if (totalPages <= 100) return 5;

    return 8;
  }

  _Range _testRangeFromChunk({
    required MemorizationScheduleChunk chunk,
    required int testIndex,
  }) {
    final total = chunk.ayahsCount;

    if (total <= 12) {
      return _Range(
        start: chunk.startGlobalAyahIndex,
        end: chunk.endGlobalAyahIndex,
      );
    }

    final wanted = total <= 40 ? math.min(total, 18) : math.min(total, 35);

    if (testIndex % 3 == 0) {
      return _Range(
        start: chunk.startGlobalAyahIndex,
        end: (chunk.startGlobalAyahIndex + wanted - 1)
            .clamp(chunk.startGlobalAyahIndex, chunk.endGlobalAyahIndex)
            .toInt(),
      );
    }

    return _Range(
      start: math.max(
        chunk.startGlobalAyahIndex,
        chunk.endGlobalAyahIndex - wanted + 1,
      ),
      end: chunk.endGlobalAyahIndex,
    );
  }

  _TestVariation _testVariation(int index) {
    final value = index % 3;

    if (value == 0) {
      return const _TestVariation(
        code: 'starts',
        subtitle: 'اختبار بدايات الآيات. الموضع يظهر عند فتح الاختبار.',
      );
    }

    if (value == 1) {
      return const _TestVariation(
        code: 'hidden',
        subtitle: 'اختبار آيات مخفية. الموضع يظهر عند فتح الاختبار.',
      );
    }

    return const _TestVariation(
      code: 'page',
      subtitle: 'اختبار صفحة كاملة. الموضع يظهر عند فتح الاختبار.',
    );
  }

  MemorizationTodayTaskModel _taskFromChunk({
    required MemorizationActivePlanModel plan,
    required MemorizationScheduleChunk chunk,
    required int index,
    required String type,
    required String title,
    required String subtitle,
    required DateTime scheduledDate,
    required String timeCode,
    required int expectedMinutes,
    int? minGlobalAyahIndex,
    int? maxGlobalAyahIndex,
  }) {
    return _taskFromRange(
      plan: plan,
      type: type,
      title: title,
      subtitle: subtitle,
      startGlobalAyahIndex: chunk.startGlobalAyahIndex,
      endGlobalAyahIndex: chunk.endGlobalAyahIndex,
      scheduledDate: scheduledDate,
      timeCode: '${timeCode}_${chunk.index}',
      expectedMinutes: expectedMinutes,
      minGlobalAyahIndex: minGlobalAyahIndex,
      maxGlobalAyahIndex: maxGlobalAyahIndex,
    );
  }

  MemorizationTodayTaskModel _taskFromRange({
    required MemorizationActivePlanModel plan,
    required String type,
    required String title,
    required String subtitle,
    required int startGlobalAyahIndex,
    required int endGlobalAyahIndex,
    required DateTime scheduledDate,
    required String timeCode,
    required int expectedMinutes,
    int? minGlobalAyahIndex,
    int? maxGlobalAyahIndex,
  }) {
    final minBound = (minGlobalAyahIndex ?? plan.scopeStartGlobalAyahIndex)
        .clamp(0, 6235)
        .toInt();

    final maxBound = (maxGlobalAyahIndex ?? plan.scopeEndGlobalAyahIndex)
        .clamp(minBound, 6235)
        .toInt();

    final safeStart = startGlobalAyahIndex.clamp(minBound, maxBound).toInt();
    final safeEnd = endGlobalAyahIndex.clamp(safeStart, maxBound).toInt();
    final cleanDate = _dateOnly(scheduledDate);

    return MemorizationTodayTaskModel(
      id: 'journey_${plan.id}_${timeCode}_${safeStart}_$safeEnd',
      planId: plan.id,
      type: type,
      title: title,
      subtitle: subtitle,
      scopeTitle: rangeResolver.titleForRange(safeStart, safeEnd),
      startGlobalAyahIndex: safeStart,
      endGlobalAyahIndex: safeEnd,
      expectedMinutes: expectedMinutes,
      isCompleted: false,
      status: MemorizationTodayTaskModel.statusNotStarted,
      scheduledDate: cleanDate,
      createdAt: cleanDate,
      updatedAt: cleanDate,
    );
  }

  int _reviewSessionsCountForPlan(MemorizationActivePlanModel plan) {
    if (plan.reviewSessionsCount > 0) return plan.reviewSessionsCount;

    if (plan.dailyReviewPages > 0) {
      final reviewStartPage = rangeResolver.pageForGlobalAyah(
        plan.reviewStartGlobalAyahIndex,
      );
      final reviewEndPage = rangeResolver.pageForGlobalAyah(
        plan.reviewEndGlobalAyahIndex,
      );
      final pages = math.max(1, reviewEndPage - reviewStartPage + 1);

      return (pages / plan.dailyReviewPages).ceil().clamp(1, 999).toInt();
    }

    if (plan.totalDays > 0) return plan.totalDays;

    return 1;
  }

  int _estimateMinutes({
    required int ayahsCount,
    required String type,
  }) {
    final multiplier = type == 'dailyNew'
        ? 1.35
        : type == 'selfTest'
        ? 0.55
        : 0.75;

    final minutes = 5 + (ayahsCount * multiplier);

    return minutes.clamp(6, 60).round();
  }

  DateTime _dateOnly(DateTime value) {
    return DateTime(value.year, value.month, value.day);
  }

  bool _sameDay(DateTime a, DateTime b) {
    return a.year == b.year && a.month == b.month && a.day == b.day;
  }
}

class _Range {
  final int start;
  final int end;

  const _Range({
    required this.start,
    required this.end,
  });

  int get ayahsCount => end - start + 1;
}

class _TestVariation {
  final String code;
  final String subtitle;

  const _TestVariation({
    required this.code,
    required this.subtitle,
  });
}
