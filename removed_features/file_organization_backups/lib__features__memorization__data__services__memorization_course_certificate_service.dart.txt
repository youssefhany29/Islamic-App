import 'dart:math' as math;

import 'package:islamic_app/core/services/user_profile_service.dart';
import 'package:islamic_app/features/memorization/test/services/memorization_test_result_storage.dart';

import '../models/memorization_active_plan_model.dart';
import '../models/memorization_course_certificate.dart';
import '../models/memorization_session_result_model.dart';
import 'memorization_journey_companion_service.dart';

class MemorizationCourseCertificateService {
  const MemorizationCourseCertificateService();

  Future<MemorizationCourseCertificate?> buildForPlan({
    required MemorizationActivePlanModel plan,
    required List<MemorizationSessionResultModel> results,
  }) async {
    final planResults = results
        .where((result) {
          if (result.completedAt.isBefore(plan.createdAt)) return false;
          return result.endGlobalAyahIndex >= plan.scopeStartGlobalAyahIndex &&
              result.startGlobalAyahIndex <= plan.scopeEndGlobalAyahIndex;
        })
        .toList(growable: false);

    final memorizedAyahs = <int>{};
    for (final result in planResults.where(
      (item) => item.taskType == 'dailyNew',
    )) {
      final start = math.max(
        plan.scopeStartGlobalAyahIndex,
        result.startGlobalAyahIndex,
      );
      final end = math.min(
        plan.scopeEndGlobalAyahIndex,
        result.endGlobalAyahIndex,
      );
      for (int index = start; index <= end; index++) {
        memorizedAyahs.add(index);
      }
    }

    final totalAyahs = math.max(1, plan.totalAyahs);
    final memorizationPercent = ((memorizedAyahs.length / totalAyahs) * 100)
        .round()
        .clamp(0, 100);
    if (memorizationPercent < 100) return null;

    final testResults = await const MemorizationTestResultStorage().getResults(
      planId: plan.id,
    );
    final testsAverage = testResults.isEmpty
        ? _averageRating(
            planResults.where((item) => item.taskType == 'selfTest'),
          )
        : (testResults.fold<double>(
                    0,
                    (sum, result) => sum + result.scorePercent,
                  ) /
                  testResults.length)
              .round()
              .clamp(0, 100);
    final reviewResults = planResults.where(
      (item) => item.taskType == 'dailyReview' || item.taskType == 'weakReview',
    );
    final reviewPercent = _averageRating(reviewResults);
    final weakResults = planResults.where(
      (item) => item.rating == 'hard' || item.rating == 'forgot',
    );
    final improvedWeakSpotsPercent = planResults.isEmpty
        ? 0
        : (100 - ((weakResults.length / planResults.length) * 100).round())
              .clamp(0, 100);
    final companion = await const MemorizationJourneyCompanionService()
        .buildReport();
    final commitmentPercent = companion.activePlan?.id == plan.id
        ? companion.commitmentPercent
        : 100;
    final finalScore =
        ((memorizationPercent * 0.35) +
                (commitmentPercent * 0.25) +
                (testsAverage * 0.25) +
                (reviewPercent * 0.15))
            .round()
            .clamp(0, 100);
    final completionDate = planResults.isEmpty
        ? DateTime.now()
        : planResults
              .map((item) => item.completedAt)
              .reduce((a, b) => a.isAfter(b) ? a : b);
    final completedDays = planResults
        .map(
          (item) =>
              '${item.completedAt.year}-${item.completedAt.month}-${item.completedAt.day}',
        )
        .toSet()
        .length;

    return MemorizationCourseCertificate(
      userName: await const UserProfileService().getUserName(),
      planName: plan.planName,
      scopeTitle: plan.scopeTitle,
      startedAt: plan.createdAt,
      completedAt: completionDate,
      commitmentPercent: commitmentPercent,
      memorizationPercent: memorizationPercent,
      testsAveragePercent: testsAverage,
      reviewPercent: reviewPercent,
      improvedWeakSpotsPercent: improvedWeakSpotsPercent,
      completedDays: completedDays,
      finalScore: finalScore,
    );
  }

  int _averageRating(Iterable<MemorizationSessionResultModel> results) {
    final items = results.toList(growable: false);
    if (items.isEmpty) return 0;
    final total = items.fold<int>(0, (sum, item) {
      return sum +
          switch (item.rating) {
            'easy' => 100,
            'good' => 75,
            'hard' => 40,
            'forgot' => 15,
            _ => 70,
          };
    });
    return (total / items.length).round().clamp(0, 100);
  }
}
